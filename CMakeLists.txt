cmake_minimum_required(VERSION 3.16)
project(mcserverkit LANGUAGES C)

if (WIN32)
    set(output "mcserverkit.dll")
    set(import_lib "${CMAKE_CURRENT_BINARY_DIR}/libmcserverkit.a")
elseif(APPLE)
    set(output "mcserverkit.dylib")
else()
    set(output "mcserverkit.so")
endif()

set(lib "${CMAKE_BINARY_DIR}/${output}")
set(h "${CMAKE_BINARY_DIR}/mcserverkit.h")

add_custom_command(OUTPUT "${lib}" "${h}" COMMAND go build -buildmode=c-shared -o "${lib}" ./bind WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}")
if (WIN32)
    add_custom_command(OUTPUT "${import_lib}" COMMAND gendef "${lib}" COMMAND ${CMAKE_COMMAND} -E rename "${CMAKE_BINARY_DIR}/mcserverkit.def" "${CMAKE_CURRENT_BINARY_DIR}/mcserverkit.def" COMMAND dlltool -d "${CMAKE_CURRENT_BINARY_DIR}/mcserverkit.def" -l "${import_lib}" DEPENDS "${lib}")
    add_custom_target(bin ALL DEPENDS "${lib}" "${h}" "${import_lib}")
else()
add_custom_target(bin ALL DEPENDS "${lib}" "${h}")
endif()

add_library(mcserverkit SHARED IMPORTED GLOBAL)

if (WIN32)
    set_target_properties(mcserverkit PROPERTIES IMPORTED_LOCATION "${lib}" IMPORTED_IMPLIB "${import_lib}" INTERFACE_INCLUDE_DIRECTORIES "${CMAKE_BINARY_DIR}")
else()
set_target_properties(mcserverkit PROPERTIES IMPORTED_LOCATION "${lib}" INTERFACE_INCLUDE_DIRECTORIES "${CMAKE_BINARY_DIR}")
endif()
